FROM students;

SELECT LOWER(CONCAT('Good Morning ', CONCAT(SUBSTR(student_firstname,1,2), SUBSTR(student_surname,1,5))))
FROM students;

SELECT CONCAT('good morning ',
CONCAT(TRIM(TRAILING ' ' FROM CONCAT(SUBSTR(student_firstname,1,1),LOWER(SUBSTR(student_firstname,2,2)))),
TRIM(TRAILING ' ' FROM CONCAT(SUBSTR(student_surname,1,1), LOWER(SUBSTR(student_surname,2,5))))))
FROM students;
