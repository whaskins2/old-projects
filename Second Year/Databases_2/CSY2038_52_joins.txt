SELECT sa.subject_id, sa.subject_name, c.title
FROM subject_areas sa
JOIN courses c
ON sa.subject_id = c.subject_id;

SELECT c.course_id, c.title, i.instructor_name, i.instructor_id
FROM courses c
JOIN offerings o
ON c.course_id = o.course_id
JOIN instructors i
ON i.instructor_id = o.instructor_id;

COLUMN title FORMAT A10;
SELECT c.course_id, c.title, c.cost, i.instructor_name, i.instructor_id
FROM courses c
JOIN offerings o
ON c.course_id = o.course_id
JOIN instructors i
ON i.instructor_id = o.instructor_id
ORDER BY c.cost;

SELECT s.student_id, s.student_firstname, s.student_surname, a.evaluation, o.offering_id
FROM students s
LEFT JOIN attendance a
ON s.student_id = a.student_id
JOIN offerings o
ON o.offering_id = a.offering_id
ORDER BY s.student_id;

SELECT c.course_id, c.title, o.offering_id, s.site_id, s.location
FROM courses c
FULL JOIN offerings o
ON c.course_id = o.course_id
FULL JOIN sites s
ON o.site_id = s.site_id
ORDER BY c.course_id;

SELECT s.site_id, s.location, sa.subject_id, sa.subject_name
FROM sites s
LEFT JOIN offerings o
ON s.site_id = o.site_id
LEFT JOIN courses c
ON o.course_id = c.course_id
LEFT JOIN subject_areas sa
ON c.subject_id = sa.subject_id
ORDER BY s.site_id;
